public class CaseDAO {
    private static final CaseDAO instance = new CaseDAO();

    private CaseDAO(){
    }

    public static CaseDAO getInstance(){
        return instance;
    }

    public Database.QueryLocator searchRecordIntegrationError(){
        Database.QueryLocator queryLocator = Database.getQueryLocator([SELECT Id, Reason, Status, 
                                                                        (SELECT Id, Case__c FROM Logs_de_Erro__r WHERE Case__c != null), 
                                                                        ErroIntegracao__c 
                                                                        FROM Case
                                                                        WHERE ErroIntegracao__c = true
                                                                        ]);
    return queryLocator;
    }
}
